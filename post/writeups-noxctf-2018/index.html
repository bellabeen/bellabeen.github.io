# return (g, x, y) a*x + b*y = gcd(x, y)
def egcd(a, b):
    if a == 0:
        return (b, 0, 1)
    else:
        g, x, y = egcd(b % a, a)
        return (g, y - (b // a) * x, x)

def decryptRSA(p,q,e,ct):

def encryptRSA(p,q,e,pt):

   encoded = encoded.zfill(length+length%2)

def mulinv(b, n):
    g, x, _ = egcd(b, n)

Program Penentu Hasil -N Bilangan C++ / C Plus Plus Geany Linux
Disini saya akan membagikan contoh program sederhana yang berfungsi untuk menentukan hasil dari bilangan n yang akan di jumlahkan dan dihasilkan daripada banyak omong mendingan kita liat contoh program nya

#include <iostream>
using namespace std;
int main () {
    int bil, x, y ,hasil_genap, hasil_ganjil, hasil_n;
    cout<<"Program sederhana Penentu Hasil n bilangan"<<endl;
    cout<<"=========================================="<<endl;
    cout<<"Masukkan -n Bilangan = ";
    cin>>y;
    hasil_ganjil = 0;
    hasil_genap = 0;
    hasil_n = 0;
    cout<<endl;
    for (x=1; x<=y; x++) {
        cout<<"Masukkan Bilangan ke n - "<<x<<" = ";
        cin>>bil;
        hasil_n = hasil_n+bil;
        if (bil%2==0) {
            hasil_genap = hasil_genap+bil;
        }
        else {
            hasil_ganjil = hasil_ganjil+bil;
        }
       
       
    }
    cout<<"Hasil jumlah N bilangan\t= "<<hasil_n<<endl;
    cout<<"Hasil jumlah bilangan genap  = "<<hasil_genap<<endl;
    cout<<"Hasil jumlah bilangan ganjil = "<<hasil_ganjil<<endl;
    return 0;
}

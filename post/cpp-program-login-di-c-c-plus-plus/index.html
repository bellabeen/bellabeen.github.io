Pada kali ini saya akan memberikan kodingan bagaimana membuat program login di cplus plus / C++ di geany linux
Langsung saja kita liat kodingan nya

#include <iostream>
#include <ctime>
#include <time.h>
using namespace std;
int main () {
    time_t waktu;
        string username, password;
        cout<<"Halaman login BN-Techno\n";
        cout<<"Username : ";cin>>username;
        cout<<"Password : ";cin>>password;
        if (username == "admin" && password == "admin") {
        cout<<"Anda Sukses Login di Waktu "<<ctime(&waktu)<<endl;
        }
        else {
            cout<<"Username atau Password Salah, Coba lagi"<<endl;
        }
       
    return 0;
}

keterangan 
username == "admin" && password == "admin"  sesuaikan dengan username dan password yang kalian inginkan
